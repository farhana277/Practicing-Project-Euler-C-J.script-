function specialPythagoreanTriplet(n) {
 let a,b,c;
 for(a=1;a<n/3;a++){
   for(b=a+1;b<n/2;b++){
     c=n-a-b;
          if(c>b && (a * a + b * b === c * c)){
            return a*b*c;
          }
   }
 }
return -1;
}

specialPythagoreanTriplet(1000);
